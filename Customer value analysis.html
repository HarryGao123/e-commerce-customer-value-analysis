#SSEK
# base on the minimum SSE principle to find the best K,which is the number of customers groups
from sklearn.cluster import KMeans
SSE = []
for i in range(1,10):
    kmeans = KMeans(n_clusters = i,random_state = 52)
    kmeans.fit(stand_data)
    SSE.append(kmeans.inertia_)
#check k-value
plt.plot(range(1,10),SSE,marker = 'o')
plt.show()

From the plot, we should set k as 4 or 5.
